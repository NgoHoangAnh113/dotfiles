#include <iostream>
using namespace std;

int** capPhat(int &n, int &m)
{
    cin >> n >> m;
    int **a = new int*[n];
    for (int i = 0; i < n; i++) {
        a[i] = new int[m];
    }
    return a;
}

void nhap(int **a, int n, int m)
{
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> *(a[i]+j);
        }
    }
}

void huyCapPhat(int **a, int n, int m)
{
    for (int i = 0; i < n ; i++) {
        delete []a[i];
    }
    delete []a;
}

void xuat(int **a, int n, int m)
{
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << *(a[i]+j) << " ";
        }
        cout << endl;
    }
}

int main()
{
    int **a, n, m;
    a = capPhat(n, m);
    nhap(a, n, m);
    xuat(a, n, m);
    huyCapPhat(a, n, m);
    return 0;
}
