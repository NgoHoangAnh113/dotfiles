#include <iostream>
using namespace std;

int* capPhat(int &n)
{
    cin >> n;
    int *a = new int[n];
    return a;
}

void nhap(int *a, int n)
{
    for (int i = 0; i < n; i++) {
        cin >> *(a+i);
    }
}

void huyCapPhat(int *a, int n)
{
    delete []a;
}

void xuat(int *a, int n)
{
    for (int i = 0; i < n; i++) {
        cout << *(a+i) << " ";
    }
}

int main()
{
    int *a, n;
    a = capPhat(n);
    nhap(a, n);
    xuat(a, n);
    huyCapPhat(a, n);
    return 0;
}
